14  /**
15   * Calculates the enemy forward prediction for a single {@link IPlayer}. Done by
16   * calculating the probabilities of the enemy location based on his valid
17   * {@link PlayerAction actions} with a certain recursive search depth. Then
18   * flood filling is performed for the minimum steps.
19   */
20  public class SingleEnemyPrediction {
21  
22  	private final Board<Cell> board;
23  
24  	private final IPlayer player;
25  
26  	private final List<FloodFillPoint> floodFillPoints = new ArrayList<>();
27  
28  	private FloatMatrix probabilities;
29  	private FloatMatrix minSteps;

31  	/**
32  	 * Creates a new {@link SingleEnemyPrediction} calculation object for the given
33  	 * {@link IPlayer}.
34  	 * 
35  	 * @param board  the {@link Board board} the {@link IPlayer player} moves on
36  	 * @param player the {@link IPlayer} the calculation is for
37  	 */
38  	public SingleEnemyPrediction(final Board<Cell> board, final IPlayer player) {
39  		this.board = board;
40  		this.player = player;
41  	}

43  	/**
44  	 * Performs the enemy prediction by recursively searching forward until a
45  	 * certain depth is reached. Further minimum steps are set with flood fill.
46  	 * 
47  	 * @param maxDepth the depth to search with
48  	 */
49  	public void doCalculation(final int maxDepth) {
50  		clearResults(maxDepth);
51  		final PredictivePlayer startPlayer = new PredictivePlayer(player);
52  		doRecursiveStep(startPlayer, 1, 1, maxDepth);
53  		floodFill();
54  	}

56  	/**
57  	 * Clears the result of the contained matrixes.
58  	 * 
59  	 * @param maxSteps the maximum amount of predicted rounds
60  	 */
61  	private void clearResults(final int maxSteps) {
62  		this.probabilities = new FloatMatrix(board.getWidth(), board.getHeight(),
63  				1 / Math.pow(PlayerAction.values().length, maxSteps + 1));
64  		this.minSteps = new FloatMatrix(board.getWidth(), board.getHeight(), Integer.MAX_VALUE);
65  	}
66  
67  	/**
68  	 * Performs a single recursive step of the calculation and calls more recursive
69  	 * steps for the generated children if the max depth is not reached.
70  	 * 
71  	 * @param player           {@link PredictivePlayer} to generate children with
72  	 * @param startProbability double probability value of the given
73  	 *                         {@link PredictivePlayer}
74  	 * @param depth            current depth of the calculation
75  	 * @param maxDepth         max depth the calculation should reach
76  	 */
77  	private void doRecursiveStep(final PredictivePlayer player, final double startProbability, final int depth,
78  			final int maxDepth) {
79  
80  		final List<PredictivePlayer> validChildren = getValidChildren(player);
81  		if (validChildren.size() == 0)
82  			return;
83  		final double probabilityFactor = 1 / validChildren.size();
84  		final double childProbability = startProbability * probabilityFactor;
85  
86  		for (final PredictivePlayer child : validChildren) {
87  
88  			for (final Point2i point : child.getShortTail()) {
89  				probabilities.add(point, childProbability);
90  				minSteps.min(point, depth);
91  			}
92  
93  			if (depth <= maxDepth)
94  				doRecursiveStep(child, childProbability, depth + 1, maxDepth);
95  			else
96  				floodFillPoints.add(new FloodFillPoint(depth, child.getPosition(), child.getSpeed()));
97  		}
98  	}

100 	/**
101 	 * Returns all valid children of a given {@link PredictivePlayer}.
102 	 * 
103 	 * @param player {@link PredictivePlayer} to generate children with
104 	 * @return {@link List} of {@link PredictivePlayer} objects
105 	 */
106 	private List<PredictivePlayer> getValidChildren(final PredictivePlayer player) {
107 		final List<PredictivePlayer> validChildren = new ArrayList<>();
108 		for (final PlayerAction action : PlayerAction.values()) {
109 			final PredictivePlayer child = new PredictivePlayer(player, action, board);
110 			if (child.isActive())
111 				validChildren.add(child);
112 		}
113 		return validChildren;
114 	}

116 	/**
117 	 * Executes flood fill for the remaining minimum steps.
118 	 */
119 	private void floodFill() {
120 		while (!floodFillPoints.isEmpty()) {
121 			final FloodFillPoint current = floodFillPoints.remove(floodFillPoints.size() - 1);
122 
123 			final List<Point2i> nextPoints = current.getPosition().vonNeumannNeighborhood();
124 
125 			for (final Point2i nextPoint : nextPoints) {
126 				if (board.isOnBoard(nextPoint)) {
127 					final FloodFillPoint next = current.next(nextPoint);
128 					if (next.getRound() < minSteps.getValue(nextPoint)) {
129 						minSteps.setValue(next.getPosition(), next.getRound());
130 						floodFillPoints.add(next);
131 					}
132 				}
133 			}
134 		}
135 	}
