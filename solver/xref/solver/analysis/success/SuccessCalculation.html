12   * Calculation Class to combine different success calculations
13   */
14  public class SuccessCalculation {
15      private Map<PlayerAction, FloatMatrix> matrixResult;
16      private ActionsRating ratingsResult;
17  
18      /**
19       * Initializes a new {@link SuccessCalculation}.
20       * 
21       * @param width  the width of the board and the resulting {@link FloatMatrix}
22       * @param height the height of the board and the resulting {@link FloatMatrix}
23       */
24      public SuccessCalculation(int width, int height) {
25          ratingsResult = new ActionsRating();
26          this.matrixResult = new EnumMap<>(PlayerAction.class);
27  
28          for (final PlayerAction action : PlayerAction.values()) {
29              matrixResult.put(action, new FloatMatrix(width, height, 0));
30          }
31      }
32  
33      /**
34       * Adds a new rating to the result.
35       * 
36       * @param action        the {@link PlayerAction initial action} the rating is
37       *                      valid for
38       * @param position      the {@link Point2i position} the rating is valid for
39       * @param successRating the new calculated successrating
40       */
41      public void add(PlayerAction action, Point2i position, double successRating) {
42          matrixResult.get(action).max(position, successRating);
43      }
44  
45      /**
46       * Combines the calculations of two {@link SuccessCalculation}.
47       * 
48       * @param other a {@link SuccessCalculation} whose results should be added to
49       *              this results
50       */
51      public void add(SuccessCalculation other) {
52          for (final PlayerAction action : PlayerAction.values()) {
53              final FloatMatrix otherMatrix = other.matrixResult.get(action);
54  
55              matrixResult.computeIfPresent(action, (k, v) -> v.max(otherMatrix));
56  
57              final FloatMatrix selfMatrix = this.matrixResult.get(action);
58              ratingsResult.setRating(action, selfMatrix.sum());
59          }
60      }
61  
62      /**
63       * Normalizes the Actionrating if necessary and returns it.
64       * 
65       * @return the normalized {@link ActionsRating}
66       */
67      public ActionsRating getRatingResult() {
68          if (ratingsResult.maxRating() != 1)
69              ratingsResult.normalize();
70  
71          return ratingsResult;
72      }
73  
74      /**
75       * Returns the {@link FloatMatrix ResultMatrix} for a specific initial action
76       * 
77       * @param action The initial action the matrix is needed for
78       * @return the {@link FloatMatrix ResultMatrix}
79       */
80      public FloatMatrix getMatrixResult(PlayerAction action) {
81          return matrixResult.get(action);
82      }
