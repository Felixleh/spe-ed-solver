42  	 * Creates a new {@link RatedPredictivePlayer} from a given
43  	 * {@link RatedPredictivePlayer} parent and an {@link PlayerAction} which is
44  	 * performed by the parent to generate the child.
45  	 * 
46  	 * @param parent        {@link RatedPredictivePlayer} parent to perform the
47  	 *                      action
48  	 * @param action        {@link PlayerAction} performed by the parent
49  	 * @param board         {@link Board} to perform the action on
50  	 * @param probabilities {@link FloatMatrix} containing enemy probabilities
51  	 * @param minSteps      {@link FloatMatrix} containing the minimum enemy steps
52  	 *                      for each point
53  	 */
54  	public RatedPredictivePlayer(final RatedPredictivePlayer parent, final PlayerAction action, final Board<Cell> board,
55  			final FloatMatrix probabilities, final FloatMatrix minSteps) {
56  		super(parent, action, board);
57  
58  		this.relativeRound = parent.getRelativeRound() + 1;
59  
60  		if (isActive()) {
61  			this.successRating = calculateSuccessRating(parent.getSuccessRating(), probabilities, minSteps,
62  					getShortTail(), relativeRound);
63  			this.cutOffRating = calculateCutOffRating(probabilities, minSteps, getShortTail(), relativeRound,
64  					successRating);
65  		} else {
66  			this.successRating = 0;
67  			this.cutOffRating = 0;
68  		}
69  	}

71  	/**
72  	 * Calculates the new success rating based on the success rating of the parent.
73  	 * 
74  	 * @param parentSuccessRating success rating of the parent
75  	 * @param probabilities       {@link FloatMatrix} with enemy probabilities to
76  	 *                            consider
77  	 * @param minSteps            {@link FloatMatrix} with the enemy minimum steps
78  	 *                            to consider
79  	 * @param shortTail           {@link Point2i points} of the last step from the
80  	 *                            parent to the child
81  	 * @param relativeRound       amount of round which are predictive
82  	 * @return new success rating for the child
83  	 */
84  	private double calculateSuccessRating(final double parentSuccessRating, final FloatMatrix probabilities,
85  			final FloatMatrix minSteps, final Collection<Point2i> shortTail, final int relativeRound) {
86  		double successFactor = 0;
87  		for (final Point2i point : shortTail) {
88  			if (relativeRound >= minSteps.getValue(point))
89  				successFactor = Math.max(successFactor, probabilities.getValue(point));
90  		}
91  		return (double) (parentSuccessRating * (1 - Math.pow(successFactor, SUCCESS_BOOST)));
92  	}

94  	/**
95  	 * Calculates the cut off rating based on the probabilities on the short tail.
96  	 * 
97  	 * @param probabilities {@link FloatMatrix} with enemy probabilities to consider
98  	 * @param minSteps      {@link FloatMatrix} with the enemy minimum steps to
99  	 *                      consider
100 	 * @param shortTail     {@link Point2i points} of the last step from the parent
101 	 *                      to the child
102 	 * @param relativeRound amount of round which are predictive
103 	 * @param successRating the probability to reach the given state
104 	 * @return new cut off rating for the child
105 	 */
106 	private double calculateCutOffRating(final FloatMatrix probabilities, final FloatMatrix minSteps,
107 			final Collection<Point2i> shortTail, final int relativeRound, final double successRating) {
108 
109 		double cutOff = 0;
110 		for (final Point2i point : shortTail) {
111 			if (relativeRound < minSteps.getValue(point)) {
112 				cutOff = Math.max(cutOff, probabilities.getValue(point));
113 			}
114 		}
115 		return cutOff * successRating;
116 	}

120 	 * children of this {@link RatedPredictivePlayer}.
121 	 * 
122 	 * @param board         {@link Board} to check for collisions
123 	 * @param probabilities {@link FloatMatrix} with enemy probabilities
124 	 * @param minSteps      {@link FloatMatrix} with the enemy minimum steps for
125 	 *                      each point
126 	 * @return {@link RatedPredictivePlayer children} with valid last steps
127 	 */
128 	public Collection<RatedPredictivePlayer> getValidChildren(final Board<Cell> board, final FloatMatrix probabilities,
129 			final FloatMatrix minSteps) {
130 
131 		final List<RatedPredictivePlayer> result = new ArrayList<>();
132 		for (final PlayerAction action : PlayerAction.values()) {
133 			final RatedPredictivePlayer child = new RatedPredictivePlayer(this, action, board, probabilities, minSteps);
134 			if (child.isActive() && child.getSuccessRating() > 0)
135 				result.add(child);
136 		}
137 		return result;
138 	}
