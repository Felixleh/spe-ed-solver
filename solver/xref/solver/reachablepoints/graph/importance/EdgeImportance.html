107 		}

113 	/**
114 	 * Returns the inverted, normalized Actionsrating. It will be calculated if it
115 	 * has not already been.
116 	 * 
117 	 * @return Normalized inverted Importance Actionsrating
118 	 */
119 	public ActionsRating getInvertedRatingResult() {
120 		if (invertedRatingResult == null) {
121 			invertedRatingResult = new ActionsRating();
122 
123 			if (initialEdgeImportance.isEmpty())
124 				return invertedRatingResult;
125 
126 			final int maxImportance = Collections.max(initialEdgeImportance.values());
127 
128 			if (maxImportance == 0)
129 				return invertedRatingResult;
130 
131 			for (final Entry<PlayerAction, ConcreteEdge> entry : initialEdges.entrySet()) {
132 				final float normalizedImportance = initialEdgeImportance.getOrDefault(entry.getValue(), 0)
133 						/ (float) maxImportance;
134 
135 				final float invertedNormalizedImportance = 1 - normalizedImportance;
136 				invertedRatingResult.setRating(entry.getKey(), invertedNormalizedImportance);
137 			}
138 		}
139 
140 		return invertedRatingResult;
141 	}

143 	/**
144 	 * Returns the inverted, normalized Actionsrating. It will be calculated if it
145 	 * has not already been.
146 	 * 
147 	 * @return Normalized inverted Importance Actionsrating
148 	 */
149 	public FloatMatrix getInvertedMatrixResult() {
150 		if (invertedMatrixResult == null) {
151 			invertedMatrixResult = new FloatMatrix(width, height);
152 
153 			if (initialEdgeImportance.isEmpty())
154 				return invertedMatrixResult;
155 
156 			final int maxImportance = Collections.max(initialEdgeImportance.values());
157 
158 			if (maxImportance == 0)
159 				return invertedMatrixResult;
160 
161 			for (final ConcreteEdge initialEdge : initialEdges.values()) {
162 				final float normalizedImportance = initialEdgeImportance.getOrDefault(initialEdge, 0)
163 						/ (float) maxImportance;
164 
165 				final float invertedNormalizedImportance = 1 - normalizedImportance;
166 				for (final Node cell : initialEdge.getPath()) {
167 					invertedMatrixResult.setValue(cell.getPosition(), invertedNormalizedImportance);
168 				}
169 			}
170 		}
171 
172 		return invertedMatrixResult;
173 	}
