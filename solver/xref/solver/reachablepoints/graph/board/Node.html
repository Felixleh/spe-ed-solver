45  	 * 
46  	 * @param direction The Players {@link PlayerDirection#PlayerDirection
47  	 *                  Direction}
48  	 * @param doJump    true for jump over the cells (when round % 6 == 0), else
49  	 *                  false
50  	 * @param speed     The Players Speed (1 &lt;= speed &gt;= 10)
51  	 * @return The Concrete Edge or null if the requested Edge is not passable
52  	 */
53  	public ConcreteEdge getEdge(PlayerDirection direction, boolean doJump, int speed) {
54  
55  		// if the edge is not available -> return null
56  		if (this.edges == null || value != CellValue.EMPTY_CELL) {
57  			return null;
58  		}
59  
60  		IEdge edge = this.edges[getIntegerIndex(direction, doJump, speed)];
61  
62  		// If any passed Node is not Empty -> return null
63  		if (edge == null) {
64  			return null;
65  		}
66  
67  		if (edge instanceof AbstractEdge) {
68  			edge = ((AbstractEdge) edge).calculatePath(graph, this, direction, doJump, speed);
69  			setEdge(direction, doJump, speed, (ConcreteEdge) edge);
70  		}
71  
72  		return (ConcreteEdge) edge;
73  	}

86  	 * 
87  	 * @param direction the {@link PlayerDirection}
88  	 * @param doJump    whether a jump is done
89  	 * @param speed     the speed
90  	 * @return edge-Array-index for the edge matching the passed parameters
91  	 */
92  	public static Integer getIntegerIndex(PlayerDirection direction, boolean doJump, int speed) {
93  		int index = direction.ordinal();
94  		index += (doJump ? 1 : 0) << PLAYER_DIRECTION_ORDINAL_HIGHEST_ONE_BIT;
95  		index += (speed - 1) << (PLAYER_DIRECTION_ORDINAL_HIGHEST_ONE_BIT + 1);
96  		return index;
97  	}

119 	 * @param removeEdgesStartingHere Specifies whether edges that start at this

121 	 *                                kept for the time being if there is currently
122 	 *                                a player there.
123 	 */
124 	public void setCellValue(CellValue value, boolean removeEdgesStartingHere) {
125 		if (this.value != value && value != CellValue.EMPTY_CELL) {
126 
127 			// Destroy all edges that pass this node or end here
128 			for (final Entry<Vector2i, int[]> edge : AffectedEdgesLookUpTable.getAbstractPassingEdges()) {
129 				Point2i edgeStartingPosition = this.position.translate(edge.getKey());
130 				Node edgeStartingNode = graph.getBoardCellAt(edgeStartingPosition);
131 
132 				if (edgeStartingNode == null)
133 					continue;
134 
135 				for (int edgeIndex : edge.getValue()) {
136 					edgeStartingNode.setEdge(edgeIndex, null);
137 				}
138 			}
139 		}
140 
141 		if (removeEdgesStartingHere && value != CellValue.EMPTY_CELL) {
142 			if (this.value != CellValue.EMPTY_CELL)
143 				this.value = CellValue.MULTIPLE_PLAYER;
144 			else
145 				this.value = value;
146 			this.edges = null;
147 		}
148 	}
