15       * points}.
16       * 
17       * @param pointA the first {@link Point2i point} of the {@link LineSegment2i
18       *               segment}
19       * @param pointB the second {@link Point2i point} of the {@link LineSegment2i
20       *               segment}
21       * @throws IllegalArgumentException if the {@link LineSegment2i segment} is not
22       *                                  vertical or horizontal
23       */
24      public LineSegment2i(Point2i pointA, Point2i pointB) {
25          this.pointA = pointA;
26          this.pointB = pointB;

28          if (pointA.getX() != pointB.getX() && pointA.getY() != pointB.getY()) {

30          }

51      /**
52       * Check for Intersection (Euclidean geometry) with another {@link LineSegment2i
53       * segment}.
54       * 
55       * @param other the other {@link LineSegment2i segment}
56       * @return true if the {@link LineSegment2i segment segments} intersect,
57       *         otherwise false
58       */
59      public boolean intersect(LineSegment2i other) {
60          if (this.isVertical() == other.isVertical()) {
61              // both vertical or horizontal
62              // intersects if one segment contains the start or endpoint of the other segment
63              return this.contains(other.pointA) || this.contains(other.pointB);
64          }
65  
66          // one vertical, one horizontal
67          LineSegment2i vertical;
68          LineSegment2i horizontal;
69          if (this.isVertical()) {
70              vertical = this;
71              horizontal = other;
72          } else {
73              vertical = other;
74              horizontal = this;
75          }
76  
77          boolean horizontalIsLeftFromVertical = horizontal.pointA.getX() <= vertical.pointA.getX()
78                  || horizontal.pointB.getX() <= vertical.pointA.getX();
79  
80          boolean horizontalIsRightFromVertical = horizontal.pointA.getX() >= vertical.pointA.getX()
81                  || horizontal.pointB.getX() >= vertical.pointA.getX();
82  
83          boolean verticalIsTopFromHorizontal = vertical.pointA.getY() >= horizontal.pointA.getY()
84                  || vertical.pointB.getY() >= horizontal.pointA.getY();
85  
86          boolean verticalIsDownFromHoriozontal = vertical.pointA.getY() <= horizontal.pointA.getY()
87                  || vertical.pointB.getY() <= horizontal.pointA.getY();
88  
89          return horizontalIsLeftFromVertical && horizontalIsRightFromVertical && verticalIsTopFromHorizontal
90                  && verticalIsDownFromHoriozontal;
91      }

93      /**
94       * Checks if the {@link LineSegment2i segment} contains a {@link Point2i}.
95       * 
96       * @param point the {@link Point2i} to check
97       * @return true if the {@link LineSegment2i segment} contains the
98       *         {@link Point2i}
99       */
100     public boolean contains(Point2i point) {
101         int aValue;
102         int bValue;
103         int pValue;
104         if (getPointA().getX() == point.getX()) {
105             // x value is the same, y value of the point must be between pointA and pointB
106             aValue = getPointA().getY();
107             bValue = getPointB().getY();
108             pValue = point.getY();
109         } else if (getPointA().getY() != point.getY()) {
110             // y value is the same, x value of the point must be between pointA and pointB
111             aValue = getPointA().getX();
112             bValue = getPointB().getX();
113             pValue = point.getX();
114         } else
115             return false;
116 
117         // check if the value of the point that is different is between pointA and
118         // pointB
119         return (aValue <= pValue && bValue >= pValue) || (aValue >= pValue && bValue <= pValue);
120     }
