6    * A line segment is a part of a vertical or horizontal line that is bounded by

26          }

37      /**
38       * Check for Intersection (Euclidean geometry) with the existing edges
39       * 
40       * @param other the other segment
41       * @return true if the segments intersect, otherwise false
42       */
43      public boolean intersect(LineSegment2i other) {
44          if (this.isVertical() == other.isVertical()) {
45              // both vertical or horizontal
46              // intersects if one segment contains the start or endpoint of the other segment
47              return this.contains(other.pointA) || this.contains(other.pointB);
48          }
49  
50          // one vertical, one horizontal
51          LineSegment2i vertical;
52          LineSegment2i horizontal;
53          if (this.isVertical()) {
54              vertical = this;
55              horizontal = other;
56          } else {
57              vertical = other;
58              horizontal = this;
59          }
60  
61          boolean horizontalIsLeftFromVertical = horizontal.pointA.getX() <= vertical.pointA.getX()
62                  || horizontal.pointB.getX() <= vertical.pointA.getX();
63  
64          boolean horizontalIsRightFromVertical = horizontal.pointA.getX() >= vertical.pointA.getX()
65                  || horizontal.pointB.getX() >= vertical.pointA.getX();
66  
67          boolean verticalIsTopFromHorizontal = vertical.pointA.getY() >= horizontal.pointA.getY()
68                  || vertical.pointB.getY() >= horizontal.pointA.getY();
69  
70          boolean verticalIsDownFromHoriozontal = vertical.pointA.getY() <= horizontal.pointA.getY()
71                  || vertical.pointB.getY() <= horizontal.pointA.getY();
72  
73          return horizontalIsLeftFromVertical && horizontalIsRightFromVertical && verticalIsTopFromHorizontal
74                  && verticalIsDownFromHoriozontal;
75      }

77      /**
78       * Checks if the segment contains a point
79       * 
80       * @param point The point to check
81       * @return true if the segment contains the point
82       */
83      public boolean contains(Point2i point) {
84          int aValue;
85          int bValue;
86          int pValue;
87          if (getPointA().getX() == point.getX()) {
88              // x value is the same, y value of the point must be between pointA and pointB
89              aValue = getPointA().getY();
90              bValue = getPointB().getY();
91              pValue = point.getY();
92          } else if (getPointA().getY() != point.getY()) {
93              // y value is the same, x value of the point must be between pointA and pointB
94              aValue = getPointA().getX();
95              bValue = getPointB().getX();
96              pValue = point.getX();
97          } else
98              return false;
99  
100         // check if the value of the point that is different is between pointA and
101         // pointB
102         return (aValue <= pValue && bValue >= pValue) || (aValue >= pValue && bValue <= pValue);
103     }
