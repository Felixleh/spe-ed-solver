18  	/**
19  	 * Logs the Data of a GameStep.
20  	 * 
21  	 * @param step the {@link GameStep Step} to log
22  	 */
23  	public static void logGameStep(GameStep step) {
24  
25  		String running;
26  		if (step.isRunning())
27  			running = "is running";
28  		else
29  			running = "finished";
30  
31  		// Log the general state
32  		StringBuilder gameStep = new StringBuilder(
33  				String.format("%s: {you=%d, round=%d, playerCount=%d, remainingMilliseconds=%s}", running,
34  						step.getSelf().getPlayerId(), step.getSelf().getRound(), step.getPlayerCount(),
35  						step.getDeadline().getRemainingMilliseconds()));
36  
37  		// Log the Player
38  		IPlayer currentPlayer = step.getSelf();
39  		final Iterator<IPlayer> iterator = step.getEnemies().values().iterator();
40  		do {
41  			String active;
42  			if (currentPlayer.isActive())
43  				active = "alive";
44  			else
45  				active = "dead";
46  
47  			String player = String.format("%n\tPlayer %d {%s, %s, %s, %d}", currentPlayer.getPlayerId(), active,
48  					currentPlayer.getDirection().name(), currentPlayer.getPosition().toString(),
49  					currentPlayer.getSpeed());
50  
51  			gameStep.append(player);
52  
53  			if (iterator.hasNext())
54  				currentPlayer = iterator.next();
55  			else
56  				currentPlayer = null;
57  		} while (currentPlayer != null);
58  
59  		final String consoleMessage = gameStep.toString();
60  
61  		// Log the Board
62  		gameStep.append(String.format("%nboard=%s", step.getBoard().toString()));
63  		final String logFileMessage = gameStep.toString();
64  
65  		ApplicationLogger.logMessage(LoggingLevel.GAME_INFO, logFileMessage, LoggingLevel.GAME_INFO, consoleMessage);
66  	}
