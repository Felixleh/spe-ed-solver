60  	@Option(names = { "-s",
61  			"--solvers" }, description = "Sets the solver types to play the game with, seperated with \",\"")
62  	public void setSolverTypes(final String solverTypesString) {
63  
64  		final List<SolverType> solverTypes = new ArrayList<>();
65  
66  		final String[] solverTypeStrings = solverTypesString.split(",");
67  		for (final String solverTypeString : solverTypeStrings) {
68  			try {
69  				solverTypes.add(SolverType.valueOf(solverTypeString));
70  			} catch (final IllegalArgumentException e) {
71  				throw new ParameterException(spec.commandLine(),
72  						String.format("The %s solver type is unknown! Valid values are %s seperated by \",\".",
73  								solverTypeString, Arrays.asList(SolverType.values())));
74  			}
75  		}
76  
77  		if (solverTypes.size() < 2)
78  			throw new ParameterException(spec.commandLine(), "You need at least 2 solvers to simulate a game!");
79  
80  		if (solverTypes.size() > 6)
81  			throw new ParameterException(spec.commandLine(), "You can't play spe_ed with more than 6 solvers!");
82  
83  		this.solverTypes = solverTypes;
84  	}

92  	@Option(names = { "-c",
93  			"--consoleLoggingLevel" }, description = "Limits the outputs in the console, a higher level includes all lower levels.\r\n"
94  					+ "ERROR = 0\r\n" + "WARNING = 1\r\n" + "GAME_INFO = 2\r\n" + "INFO = 3\r\n", defaultValue = "3")
95  	public void setConsoleOutputMethod(final String loggingLevel) {
96  		try {
97  			// Try parsing to an integer
98  			int ordinalLevel = Integer.parseInt(loggingLevel);
99  			if (ordinalLevel >= LoggingLevel.ERROR.getLevel() && ordinalLevel <= LoggingLevel.INFO.getLevel()) {
100 				ApplicationLogger.setConsoleLoggingLevel(LoggingLevel.fromInteger(ordinalLevel));
101 				return;
102 			}
103 		} catch (IllegalArgumentException e) {
104 			// try getting the Level by name
105 			for (LoggingLevel level : LoggingLevel.values()) {
106 				if (level.name().equals(loggingLevel)) {
107 					ApplicationLogger.setConsoleLoggingLevel(level);
108 					return;
109 				}
110 			}
111 		}
112 
113 		throw new ParameterException(spec.commandLine(), loggingLevel + " Is not a valid logging level!");
114 	}
