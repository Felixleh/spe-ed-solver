37  	 * Initilizes a new Simulated Spe-ed Game.
38  	 * 
39  	 * @param height      height of the Board
40  	 * @param width       widht of the Board
41  	 * @param playerCount number of Simulated Players
42  	 */
43  	public Game(final int height, final int width, final int playerCount) {
44  		final Cell[][] cells = new Cell[height][width];
45  		for (int row = 0; row < height; row++)
46  			for (int col = 0; col < width; col++)
47  				cells[row][col] = new Cell(0);
48  
49  		this.board = new Board<>(cells);
50  		this.players = new SimulationPlayer[playerCount];
51  		this.random = new Random();
52  		this.deadline = new SimulationDeadline();
53  	}

55  	/**
56  	 * Starts the {@link Game simulated Game}.
57  	 * 
58  	 * @return initial {@link GameStep} for each {@link SimulationPlayer}
59  	 */
60  	public List<GameStep> startSimulation() {
61  		// Initialize Players with a random startposition and a random direction
62  		final Set<Point2i> notAvailableStartPositions = new HashSet<>();
63  		for (int playerIndex = 0; playerIndex < players.length; playerIndex++) {
64  
65  			Point2i randomStartPosition;
66  			do {
67  				final int randomX = random.nextInt(board.getWidth());
68  				final int randomY = random.nextInt(board.getHeight());
69  				randomStartPosition = new Point2i(randomX, randomY);
70  			} while (notAvailableStartPositions.contains(randomStartPosition));
71  
72  			// Add the new positions neighbors and the neighbor-neighbors to the not allowed
73  			// start-points
74  			notAvailableStartPositions.addAll(randomStartPosition.vonNeumannNeighborhood().stream()
75  					.flatMap(l -> l.vonNeumannNeighborhood().stream()).collect(Collectors.toSet()));
76  
77  			PlayerDirection randomStartDirection = EnumExtensions.getRandomValue(PlayerDirection.class);
78  
79  			final int playerId = playerIndex + 1;
80  			this.players[playerIndex] = new SimulationPlayer(playerId, randomStartPosition, randomStartDirection,
81  					SimulationPlayer.MIN_SPEED, 0);
82  			setCell(randomStartPosition, playerId);
83  		}
84  
85  		deadline.resetDeadLine();
86  
87  		return generateGameSteps();
88  	}

90  	/**
91  	 * Applies an {@link PlayerAction action} to a {@link SimulationPlayer}. The
92  	 * {@link SimulationPlayer player} dies if this is the second
93  	 * {@link PlayerAction action action} in the same round. When all Players
94  	 * applied an {@link PlayerAction action action}, a new round will start.
95  	 * 
96  	 * @param playerId the Players Id (1 - 6)
97  	 * @param action   the action to be applied
98  	 * @return the next Game-State, if all Players sent an {@link PlayerAction
99  	 *         action} for this round. Else {@code null}
100 	 */
101 	public List<GameStep> setAction(final int playerId, final PlayerAction action) {
102 		final SimulationPlayer currentPlayer = this.players[playerId - 1];
103 
104 		currentPlayer.setNextAction(action);
105 
106 		// End round, if all players sent an Action
107 		if (Arrays.stream(players).allMatch(SimulationPlayer::isReadyToMove)) {
108 
109 			Arrays.stream(players).forEach(SimulationPlayer::doActionAndMove);
110 			updateGameState();
111 			deadline.resetDeadLine();
112 			return generateGameSteps();
113 		} else {
114 			// return new list, because no new gameState is available
115 			return new ArrayList<>();
116 		}
117 	}

119 	/**
120 	 * Generates for each {@link SimulationPlayer} a {@link GameStep gameStep
121 	 * object}.
122 	 * 
123 	 * @return a {@link List} of {@link GameStep GameSteps}
124 	 */
125 	private List<GameStep> generateGameSteps() {
126 		final List<GameStep> gameSteps = new ArrayList<>();
127 
128 		// is running when more then one player is active
129 		boolean isRunning = Arrays.stream(players).filter(SimulationPlayer::isActive).count() > 1;
130 
131 		long remainingDeadlineMilliseconds = deadline.getRemainingMilliseconds();
132 		for (final SimulationPlayer player : this.players) {
133 			final Map<Integer, IPlayer> enemies = Arrays.stream(this.players).filter(p -> p != player)
134 					.collect(Collectors.toMap(SimulationPlayer::getPlayerId, IPlayer.class::cast));
135 
136 			final PlayerDeadline individualDeadline = new PlayerDeadline(remainingDeadlineMilliseconds);
137 
138 			gameSteps.add(new GameStep((IPlayer) player, enemies, individualDeadline, board, isRunning));
139 		}
140 
141 		return gameSteps;
142 	}

144 	/**
145 	 * Applies the did steps to the Board, processes collisions and increse round.
146 	 */
147 	private void updateGameState() {
148 		HashMap<Point2i, Integer> newPassedCells = new HashMap<>();
149 
150 		for (final SimulationPlayer player : this.players) {
151 			if (!player.isActive())
152 				continue;
153 
154 			// calculate the Vector to the first step by moving back (speed - 1) steps
155 			final Vector2i firstStepVector = player.getDirection().getDirectionVector()
156 					.multiply((player.getSpeed() - 1) * -1);
157 			final Point2i firstStepPosition = player.getPosition().translate(firstStepVector);
158 
159 			applyPassedSteps(player, firstStepPosition, newPassedCells);
160 		}
161 
162 		round++;
163 	}

165 	/**
166 	 * Applies the last steps from a {@link SimulationPlayer} to the Board. The
167 	 * {@link SimulationPlayer} dies, if he causes a collision.
168 	 * 
169 	 * @param player               the player whose steps are to be applied
170 	 * @param oldPosition          the players position at the start of this round
171 	 * @param passedCellsThisRound Cells that have already been used by a player in
172 	 *                             this round
173 	 */
174 	private void applyPassedSteps(final SimulationPlayer player, final Point2i oldPosition,
175 			final HashMap<Point2i, Integer> passedCellsThisRound) {
176 
177 		final List<Point2i> passedSteps = getPassedSteps(oldPosition, player.getPosition());
178 
179 		for (final Point2i step : passedSteps) {
180 			if (setCell(step, player.getPlayerId()) == CellValue.MULTIPLE_PLAYER) {
181 				player.die();
182 			}
183 
184 			passedCellsThisRound.computeIfPresent(step, (key, value) -> {
185 				player.die();
186 				if (value > 0)
187 					players[value - 1].die();
188 				return -1;
189 			});
190 
191 			passedCellsThisRound.computeIfAbsent(step, k -> player.getPlayerId());
192 
193 		}
194 	}

196 	/**
197 	 * Calculates the passed Cells between two positions.
198 	 * 
199 	 * @param positionA the first {@link Point2i position}
200 	 * @param positionB the second {@link Point2i position}
201 	 * @return the passed {@link Point2i cell positions} (Jumped over cells are
202 	 *         excluded)
203 	 */
204 	private List<Point2i> getPassedSteps(final Point2i positionA, final Point2i positionB) {
205 		if (positionA.equals(positionB))
206 			return Arrays.asList(positionA);
207 		else if (round % JUMP_FREQUENCY == 0)
208 			return Arrays.asList(positionA, positionB);
209 		else
210 			return positionB.pointsInRectangle(positionA);
211 	}

213 	/**
214 	 * Sets a Board Cell with a value (playerId) and returns the new value.
215 	 * 
216 	 * @param point    The {@link Point2i} to set
217 	 * @param playerId The new playerId to set
218 	 * @return The actual new {@link CellValue value}
219 	 */
220 	private CellValue setCell(final Point2i point, final int playerId) {
221 		if (!board.isOnBoard(point))
222 			return CellValue.MULTIPLE_PLAYER;
223 		else if (board.getBoardCellAt(point).getCellValue() == CellValue.EMPTY_CELL)
224 			board.getBoardCellAt(point).setCellValue(playerId);
225 		else
226 			board.getBoardCellAt(point).setCellValue(CellValue.MULTIPLE_PLAYER.getIntegerValue());
227 
228 		return board.getBoardCellAt(point).getCellValue();
229 	}
