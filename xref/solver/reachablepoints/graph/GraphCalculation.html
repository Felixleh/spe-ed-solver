70

104 	/**
105 	 * Adds a {@link RatedPredictiveGraphPlayer player} to the correct
106 	 * {@link GraphCalculation#queues queue}
107 	 * 
108 	 * @param player The {@link RatedPredictiveGraphPlayer player} to add
109 	 */
110 	public void addPlayerToQueue(RatedPredictiveGraphPlayer player) {
111 		final PlayerAction action = player.getInitialAction();
112 		final Point2i position = player.getPosition();
113 
114 		successCalculation.add(action, position, player.getSuccessRating());
115 		cutOffCalculation.add(action, position, player.getCutOffRating());
116 
117 		queues.get(player.getInitialAction()).add(player);
118 		calculatedPathsCount++;
119 	}

121 	/**
122 	 * Starts the execution for the {@link GraphCalculation} with the given
123 	 * startPlayers. Ends when no steps can be found or the Deadline exceeds the
124 	 * {@link GraphCalculation#DEADLINE_MILLISECOND_BUFFER}
125 	 */
126 	public void executeDeadline() {
127 		while (queuesHasNext() && deadline.getRemainingMilliseconds() > DEADLINE_MILLISECOND_BUFFER) {
128 			executeStep();
129 		}
130 	}

132 	/**
133 	 * Calculates the valid Children for the next execition step and adds them to
134 	 * the queue
135 	 */
136 	public void executeStep() {
137 		final RatedPredictiveGraphPlayer parent = queuesPoll();
138 		final List<RatedPredictiveGraphPlayer> children = parent.getValidChildren(graph, probabilities, minSteps);
139 
140 		for (final RatedPredictiveGraphPlayer child : children) {
141 			edgeImportance.add(parent.getInitialEdgeIncrements());
142 			addPlayerToQueue(child);
143 		}
144 	}

170 	/**
171 	 * Returns the next value for a random {@link GraphCalculation#queues queue}.
172 	 * 
173 	 * @return the next {@link RatedPredictiveGraphPlayer}
174 	 */
175 	public RatedPredictiveGraphPlayer queuesPoll() {
176 
177 		if (queues.isEmpty())
178 			throw new NoSuchElementException("Tried to poll from an empty list!");
179 
180 		final int randomActionIndex = new Random().nextInt(queuedActions.length);
181 		final PlayerAction randomAction = queuedActions[randomActionIndex];
182 
183 		if (queues.get(randomAction).hasNext())
184 			return queues.get(randomAction).poll();
185 		else {
186 			queues.remove(randomAction);
187 			queuedActions = queues.keySet().toArray(new PlayerAction[queues.size()]);
188 			return queuesPoll();
189 		}
190 	}
