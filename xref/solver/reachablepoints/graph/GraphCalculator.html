52  	@Override
53  	public void performCalculation(final GameStep gameStep, final FloatMatrix probabilities,
54  			final FloatMatrix minSteps) {
55  		this.enemyProbabilitiesMatrix = probabilities;
56  		this.enemyMinStepsMatrix = minSteps;
57  
58  		updateGraph(gameStep);
59  
60  		final List<RatedPredictiveGraphPlayer> startPlayers = RatedPredictiveGraphPlayer
61  				.getValidChildren(gameStep.getSelf(), graph, probabilities, minSteps);
62  
63  		final List<GraphCalculation> calculations = getCalculations(startPlayers, gameStep.getDeadline(), graph);
64  
65  		calculate(calculations);
66  		addResults(calculations);
67  	}

69  	/**
70  	 * Updates the available {@link IEdge edges} from the
71  	 * {@link GraphCalculator#graph graph}.
72  	 * 
73  	 * @param gameStep the new {@link GameStep game step} to apply on the graph
74  	 */
75  	private void updateGraph(GameStep gameStep) {
76  		if (this.graph == null) {
77  			// Initialize the graph with an empty Node-Array
78  			final Node[][] emptyNodes = new Node[gameStep.getBoard().getHeight()][gameStep.getBoard().getWidth()];
79  			this.graph = new Graph(emptyNodes);
80  
81  			// Initialize the alive player
82  			this.activeEnemiesIds = gameStep.getEnemies().keySet().stream().mapToInt(Integer::intValue).toArray();
83  		}
84  
85  		// Merge all players that were active last round into one list
86  		List<IPlayer> enemies = new ArrayList<>();
87  		for (int playerId : this.activeEnemiesIds)
88  			enemies.add(gameStep.getEnemies().get(playerId));
89  
90  		// Transfer the new occupied cells to the graph
91  		graph.updateGraph(gameStep.getBoard(), gameStep.getSelf(), enemies);
92  
93  		// Determine the currently dead players to ignore them in the following round
94  		// for the graph-update.
95  		this.activeEnemiesIds = enemies.stream().filter(IPlayer::isActive).mapToInt(IPlayer::getPlayerId).toArray();
96  	}

98  	/**
99  	 * Generates {@link GraphCalculation GraphCalculations} for different Threads
100 	 * with a base of {@link RatedPredictiveGraphPlayer start players}.
101 	 * 
102 	 * @param startPlayers {@link RatedPredictiveGraphPlayer players} to start the
103 	 *                     {@link GraphCalculation calculations} with
104 	 * @param deadline     {@link IDeadline} which must not be exceeded
105 	 * @param graph        the {@link Graph graph board} to find the
106 	 *                     {@link ConcreteEdge edges}
107 	 * @return the created {@link GraphCalculation calculation objects}
108 	 */
109 	private List<GraphCalculation> getCalculations(final List<RatedPredictiveGraphPlayer> startPlayers,
110 			final IDeadline deadline, final Board<Node> graph) {
111 
112 		final int width = graph.getWidth();
113 		final int height = graph.getHeight();
114 		successCalculation = new SuccessCalculation(width, height);
115 		cutOffCalculation = new CutOffCalculation(width, height);
116 		importanceCalculation = new EdgeImportance(width, height, startPlayers);
117 
118 		int threadCount = Math.min(maxThreadCount, (int) (deadline.getRemainingMilliseconds() / 500));
119 
120 		// Create a Calculation for each thread
121 		List<GraphCalculation> calculations = new ArrayList<>();
122 
123 		if (threadCount <= 1) {
124 			GraphCalculation calculation = new GraphCalculation(graph, this.enemyProbabilitiesMatrix,
125 					this.enemyMinStepsMatrix, importanceCalculation.getInitialEdges(), deadline);
126 			calculations.add(calculation);
127 			startPlayers.stream().forEach(calculation::addPlayerToQueue);
128 			return calculations;
129 		}
130 
131 		while (calculations.size() < threadCount)
132 			calculations.add(new GraphCalculation(graph, this.enemyProbabilitiesMatrix, this.enemyMinStepsMatrix,
133 					importanceCalculation.getInitialEdges(), deadline));
134 
135 		// Define the number of required start players
136 		final int threadBase = (graph.getHeight() + graph.getWidth()) * 10;
137 		final int totalBase = threadBase * threadCount;
138 
139 		// create a Base of Player states
140 		GraphCalculation baseCalculation = new GraphCalculation(graph, this.enemyProbabilitiesMatrix,
141 				this.enemyMinStepsMatrix, importanceCalculation.getInitialEdges(), deadline, totalBase);
142 		startPlayers.stream().forEach(baseCalculation::addPlayerToQueue);
143 
144 		while (baseCalculation.queuesHasNext() && baseCalculation.queuesRemaining() < totalBase) {
145 			baseCalculation.executeStep();
146 		}
147 		addResults(baseCalculation);
148 
149 		for (int calculationIndex = 0; baseCalculation
150 				.queuesHasNext(); calculationIndex = (calculationIndex + 1) % threadCount) {
151 			final RatedPredictiveGraphPlayer startPlayer = baseCalculation.queuesPoll();
152 			calculations.get(calculationIndex).addPlayerToQueue(startPlayer);
153 		}
154 
155 		return calculations;
156 	}

158 	/**
159 	 * Calculates the first given {@link GraphCalculation} in this Thread and all
160 	 * other {@link GraphCalculation calculations} in seperate Threads.
161 	 * 
162 	 * @param calculations {@link GraphCalculation calculations} that must be
163 	 *                     calculated
164 	 */
165 	private static void calculate(final List<GraphCalculation> calculations) {
166 		if (calculations.isEmpty())
167 			return;
168 
169 		final List<Thread> threads = new ArrayList<>();
170 
171 		for (int i = 1; i < calculations.size(); i++) {
172 			final GraphCalculation calculation = calculations.get(i);
173 			final Thread thread = new Thread(calculation::executeDeadline);
174 			threads.add(thread);
175 			thread.start();
176 		}
177 
178 		calculations.get(0).executeDeadline();
179 
180 		// To be on the safe side interrupt all other threads
181 		threads.stream().forEach(Thread::interrupt);
182 	}

184 	/**
185 	 * Updates all the locally stored results by collecting all result of the given
186 	 * {@link GraphCalculation calculations}.
187 	 * 
188 	 * @param calculations {@link GraphCalculation calculations} to add to the
189 	 *                     result
190 	 */
191 	private void addResults(final List<GraphCalculation> calculations) {
192 		calculations.stream().forEach(this::addResults);
193 
194 		GameLogger.logGameInformation(String.format("Calculated %d reachable points paths!", calculatedPaths));
195 		calculatedPaths = 0;
196 	}
197 
198 	/**
199 	 * Updates all the locally stored results by collecting all result of the given
200 	 * {@link GraphCalculation calculation}.
201 	 * 
202 	 * @param calculation {@link GraphCalculation calculation} to add to the result
203 	 */
204 	private void addResults(GraphCalculation calculation) {
205 		this.successCalculation.add(calculation.getSuccessCalculation());
206 		this.cutOffCalculation.add(calculation.getCutOffCalculation());
207 		this.importanceCalculation.add(calculation.getEdgeImportance());
208 		calculatedPaths += calculation.getCalculatedPathsCount();
209 	}

219 	public ActionsRating combineActionsRating(float aggressiveWeight, float defensiveWeight) {
220 		return successCalculation.getRatingResult().combine(cutOffCalculation.getRatingResult(), aggressiveWeight)
