12  /**
13   * Helper class to calculate {@link ReachablePointsSingleThreaded}. The object
14   * is representing the predictive steps performed on an initial
15   * {@link RatedPredictivePlayer} and their success and cutoff
16   * {@link FloatMatrix} results. The results are gradually build by calling
17   * {@link GradualReachablePointsCalculation#performSingleStep()}.
18   */
19  public class GradualReachablePointsCalculation {
20  
21  	private static final int QUEUE_SIZE = 10000;
22  
23  	private final Board<Cell> board;
24  
25  	private final FloatMatrix probabilities;
26  	private final FloatMatrix minSteps;
27  
28  	private final LimitedQueue<RatedPredictivePlayer> queue;
29  
30  	private final FloatMatrix successMatrixResult;
31  	private final FloatMatrix cutOffMatrixResult;
32  
33  	private int calculatedPathsCount = 0;
34  
35  	/**
36  	 * Creates a new {@link GradualReachablePointsCalculation} object with the given
37  	 * information for the initial {@link RatedPredictivePlayer}.
38  	 * 
39  	 * @param board         {@link Board} to check for collisions
40  	 * @param probabilities probabilities of enemies as {@link FloatMatrix}
41  	 * @param minSteps      minimum steps of enemies as {@link FloatMatrix}
42  	 * @param startPlayer   initial {@link RatedPredictivePlayer} all children paths
43  	 *                      depend on
44  	 */
45  	public GradualReachablePointsCalculation(final Board<Cell> board, final FloatMatrix probabilities,
46  			final FloatMatrix minSteps, final RatedPredictivePlayer startPlayer) {
47  
48  		this.board = board;
49  		this.probabilities = probabilities;
50  		this.minSteps = minSteps;
51  
52  		this.queue = new LimitedQueue<RatedPredictivePlayer>(RatedPredictivePlayer.class, QUEUE_SIZE);
53  
54  		this.successMatrixResult = new FloatMatrix(board.getWidth(), board.getHeight(), 0);
55  		this.cutOffMatrixResult = new FloatMatrix(board.getWidth(), board.getHeight(), 0);
56  
57  		if (startPlayer.isActive()) {
58  			queue.add(startPlayer);
59  			successMatrixResult.max(startPlayer.getPosition(), startPlayer.getSuccessRating());
60  			cutOffMatrixResult.max(startPlayer.getPosition(), startPlayer.getCutOffRating());
61  		}
62  	}
63  
64  	/**
65  	 * Performs a single step of the player specific reachable points calculating by
66  	 * polling a {@link RatedPredictivePlayer} from a {@link LimitedQueue},
67  	 * evaluating the {@link RatedPredictivePlayer children} and adding the valid
68  	 * {@link RatedPredictivePlayer children} to the {@link LimitedQueue}. The local
69  	 * results are added to the global result {@link FloatMatrix matrices}.
70  	 */
71  	public void performSingleStep() {
72  
73  		final RatedPredictivePlayer calculationPlayer = queue.poll();
74  		final Collection<RatedPredictivePlayer> children = calculationPlayer.getValidChildren(board, probabilities,
75  				minSteps);
76  
77  		for (final RatedPredictivePlayer child : children) {
78  			final Point2i position = child.getPosition();
79  
80  			successMatrixResult.max(position, child.getSuccessRating());
81  			cutOffMatrixResult.max(position, child.getCutOffRating());
82  
83  			queue.add(child);
84  			calculatedPathsCount++;
85  		}
86  	}
87  
88  	/**
89  	 * Determines if there is nothing more to be calculated.
90  	 * 
91  	 * @return true if nothing can be calculated anymore
92  	 */
93  	public boolean isFinished() {
94  		return queue.isEmpty();
95  	}
