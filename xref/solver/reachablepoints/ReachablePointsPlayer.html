13  /**
14   * Player implementing {@link ISpeedSolverPlayer} used for multiple
15   * {@link PlayerType} values. For each possible {@link PlayerAction} the
16   * reachable points are determined. For each reachable point is rated for its
17   * success and for the cut off potential. The {@link ActionsRating} is used to
18   * give each analysis its weight. The overall highest rated {@link PlayerAction}
19   * gets chosen for each {@link GameStep}.
20   */
21  public class ReachablePointsPlayer implements ISpeedSolverPlayer {
22  
23  	private final EnemyProbabilityCalculator enemyProbabilityCalculator;
24  	private final IReachablePoints reachablePointsCalculator;
25  	private final float aggressiveWeight;
26  	private final float defensiveWeight;
27  
28  	/**
29  	 * Creates a new {@link ReachablePointsPlayer} with the given configuration
30  	 * values.
31  	 * 
32  	 * @param enemySearchDepth recursive search depth to search for enemy actions
33  	 * @param aggressiveWeight relative weight for aggressive {@link ActionsRating}
34  	 * @param defensiveWeight  relative weight for the defensive
35  	 *                         {@link ActionsRating}
36  	 * @param type             the {@link ReachablePointsType} of the calculation
37  	 */
38  	public ReachablePointsPlayer(final int enemySearchDepth, float aggressiveWeight, float defensiveWeight,
39  			final ReachablePointsType type) {
40  		this.enemyProbabilityCalculator = new EnemyProbabilityCalculator(enemySearchDepth);
41  		this.reachablePointsCalculator = type.newInstance();
42  		this.aggressiveWeight = aggressiveWeight;
43  		this.defensiveWeight = defensiveWeight;
44  	}
45  
46  	@Override
47  	public PlayerAction calculateAction(GameStep gameStep, Consumer<ContextualFloatMatrix> boardRatingConsumer) {
48  		if (!gameStep.getSelf().isActive())
49  			return PlayerAction.CHANGE_NOTHING;
50  
51  		// Calculate enemyProbability
52  		enemyProbabilityCalculator.performCalculation(gameStep.getEnemies().values(), gameStep.getBoard());
53  
54  		// Calculate the Action
55  		reachablePointsCalculator.performCalculation(gameStep, enemyProbabilityCalculator.getProbabilitiesMatrix(),
56  				enemyProbabilityCalculator.getMinStepsMatrix());
57  
58  		// Combine the results
59  		final ActionsRating combinedActionsRating = reachablePointsCalculator.combineActionsRating(aggressiveWeight,
60  				defensiveWeight);
61  
62  		// Log the results
63  		reachablePointsCalculator.logGameInformation(combinedActionsRating);
64  
65  		// get the best action
66  		final PlayerAction actionToTake = combinedActionsRating.maxAction();
67  
68  		// send the Data to the viewer
69  		for (final ContextualFloatMatrix matrix : reachablePointsCalculator.getContextualFloatMatrices(actionToTake)) {
70  			boardRatingConsumer.accept(matrix);
71  		}
72  
73  		// send the Calculated Action
74  		return actionToTake;
75  	}
