34          final PredictivePlayer self = new PredictivePlayer(gameStep.getSelf());
35  
36          return Arrays.stream(PlayerAction.values())
37                  .filter(action -> new PredictivePlayer(self, action, gameStep.getBoard()).isActive()).collect(shuffle())
38                  .stream().findFirst().orElse(PlayerAction.CHANGE_NOTHING);
39      }
